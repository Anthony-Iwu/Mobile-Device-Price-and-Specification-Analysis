import pandas as pd
import matplotlib.pyplot as plt

# Load dataset
df = pd.read_csv(r'Mobiles Dataset (2025).csv', encoding='latin1')

# Filter flagship models (iPhone 16 Pro Max, Galaxy S24 Ultra)
flagships = df[df['Model Name'].str.contains('iPhone 16 Pro Max|Galaxy S24 Ultra')]

# Select relevant columns
price_cols = ['Model Name', 'Launched Price (Pakistan)', 'Launched Price (India)', 
              'Launched Price (China)', 'Launched Price (USA)', 'Launched Price (Dubai)']
flagship_prices = flagships[price_cols].copy()

# Clean currency symbols and convert to numeric
for col in price_cols[1:]:
   flagship_prices[col] = flagship_prices[col].str.replace(r'[^\d.]', '', regex=True).astype(float)

# Plot
flagship_prices.set_index('Model Name').plot(kind='bar', figsize=(12, 6))
plt.title('Flagship Model Prices by Region (2024)')
plt.ylabel('Price (USD Equivalent)')
plt.xticks(rotation=45)
plt.savefig("Flagship Model Prices by Region (2024).png")
plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Clean Price (USA) column
df['Price (USD)'] = df['Launched Price (USA)'].str.replace(r'[^\d.]', '', regex=True).astype(float)

# Calculate yearly trends with raw strings
yearly_trends = df.groupby('Launched Year').agg({
    'Price (USD)': 'mean',
    'Battery Capacity': lambda x: x.str.replace(r'[^\d.]', '', regex=True).astype(float).mean(),
    'RAM': lambda x: x.str.extract(r'(\d+)').astype(float).mean()
})

# Plot normalized trends
yearly_trends_norm = (yearly_trends - yearly_trends.min()) / (yearly_trends.max() - yearly_trends.min())
yearly_trends_norm.plot(kind='line', figsize=(10, 6), marker='o')
plt.title('Normalized Component Cost vs. Price Trends (20202024)')
plt.ylabel('Normalized Value')
plt.xlabel('Year')
plt.savefig("Cost vs. Price Trends.png")
plt.show()

# Create a proper copy of the dataframe
pro_models = df[df['Model Name'].str.contains('Pro', case=False)].copy()

# Define exchange rates (example values - update with real rates)
exchange_rates = {
    'China': 0.14,    # CNY to USD
    'Dubai': 0.27,    # AED to USD
    'Pakistan': 0.0036, # PKR to USD
    'USA': 1.0        # USD to USD
}

# Process prices for all regions
regions = ['USA', 'China', 'Dubai', 'Pakistan']

for region in regions:
    # Clean and convert prices using raw strings and safe conversion
    cleaned_prices = pd.to_numeric(
        pro_models[f'Launched Price ({region})'].str.replace(r'[^\d.]', '', regex=True),
        errors='coerce'
    )
    
    # Convert to USD
    pro_models[f'Price (USD) - {region}'] = cleaned_prices * exchange_rates[region]

# Optional: Verify results
print(pro_models[[f'Price (USD) - {r}' for r in regions]].describe())
